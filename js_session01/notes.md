operadores lógicos
&&, ||, ==, !=, ===, !==, !, >, <, >=, <=

var speed = 10.50;
var age = 10;

console.log(typeof 1 == typeof 1);
console.log(speed > 11 && age < 11);
console.log(speed > 11 || age < 11);
console.log(speed == age);
console.log(speed === age);
console.log(speed !== age);
console.log(!(age > 18));
